$.when(kendo.ui.ExtInputDialog.show({ 
    title: "Input", 
    message: "Please enter your name:", 
    required: true })
).done(function (response) {
    if (response.button == "OK") {
        $("#userResponse").append(kendo.format("<div>{0} button clicked at: {1} with input of: {2}</div>", 
            response.button, new Date().toLocaleTimeString(), response.input));
    } else {
        $("#userResponse").append(kendo.format("<div>{0} button clicked at: {1}</div>", 
            response.button, new Date().toLocaleTimeString()));
    }
});
